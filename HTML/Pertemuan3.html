class Ball():
    def __init__(self, y, radius, window_width, window_height, window):
        self.x = random.randint(75, window_width - 75)
        self.y = y
        self.r = radius
        self.window_width = window_width 
        self.window_height = window_height
        self.window = window
        if random.randint(0, 1) == 1:
            self.dx = -5
        else:
            self.dx = 5
        self.dy = -5
        self.rect = Rect(self.x - self.r/2, self.y - self.r/2, self.r*2, self.r*2)
    def update(self, prect):
        draw.circle(self.window, (0, 0, 0), (self.x, self.y), self.r + 3)
        draw.circle(self.window, (255, 255, 255), (self.x, self.y), self.r)
        self.x += self.dx
        self.y += self.dy
        self.rect = Rect(self.x - self.r/2, self.y - self.r/2, self.r*2, self.r*2)
        if self.x < 0 or self.x > self.window_width:
            self.dx *= -1
        if self.y < 0 or self.y > self.window_height or self.rect.colliderect(prect):
            self.dy *= -1
